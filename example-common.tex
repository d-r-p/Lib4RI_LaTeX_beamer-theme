\def\thelfrilecturepart{\alph{lfrilecturepart}}

  \ifnum\value{lfrilecturepart}>0\relax%
    \thelfrilecturepart%
  \fi%
}

\makeatletter
\newcommand\framesectiontitle{%
  \begingroup%
    \def\@sectiontitle{}%
    \ifnum\value{section}=0\relax%
    \else%
      \def\@sectiontitle{\thesection.~\insertsection}%
    \fi%
    {\@sectiontitle}%
  \endgroup%
}
\newcommand\framesectionsubtitle{%
  \begingroup%
    \def\@subsectiontitle{}%
    \def\@subsubsectiontitle{}%
    \ifnum\value{subsection}=0\relax%
    \else%
      \def\@subsectiontitle{\thesubsection.~\insertsubsection}%
    \fi%
    \ifnum\value{subsubsection}=0\relax%
    \else%
      \def\@subsubsectiontitle{~--~\scriptsize \thesubsubsection.~\insertsubsubsection}%
    \fi%
    {\@subsectiontitle\@subsubsectiontitle}%
  \endgroup%
}
\makeatother
